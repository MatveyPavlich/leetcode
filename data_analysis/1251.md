```
import pandas as pd

def average_selling_price(prices: pd.DataFrame, units_sold: pd.DataFrame) -> pd.DataFrame:
    df = prices.merge(units_sold, on="product_id", how="left")

    df = df[
        (
            (df["purchase_date"] >= df["start_date"]) &
            (df["purchase_date"] <= df["end_date"])
        )
        | df["purchase_date"].isna()
    ]

    df["units"] = df["units"].fillna(0)
    df["total"] = df["price"] * df["units"]

    agg = (
        df.groupby("product_id")
          .agg(total=("total", "sum"), units=("units", "sum"))
          .reset_index()
    )

    agg["average_price"] = 0
    mask = agg["units"] != 0
    agg.loc[mask, "average_price"] = (
        agg.loc[mask, "total"] / agg.loc[mask, "units"]
    ).round(2)

    return agg[["product_id", "average_price"]]
```
