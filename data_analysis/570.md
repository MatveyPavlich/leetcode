WITH magager_reports AS (
   SELECT
        managerId,
        COUNT(managerId) AS n_reports

    FROM employee
    GROUP BY managerId
),

final AS (
    SELECT
        ee.name
    FROM employee                   AS ee
         INNER JOIN magager_reports AS me ON ee.id = me.managerId
    WHERE me.n_reports >= 5
)

SELECT * FROM final
```

import pandas as pd

def find_managers(employee: pd.DataFrame) -> pd.DataFrame:
    df_managers =(
        employee[["managerId"]]
            .groupby(["managerId"])
            .agg(n_reports=("managerId", "count"))
            .reset_index()
    )

    df_managers = df_managers[df_managers["n_reports"] >= 5]

    df_final = pd.merge(df_managers, employee, left_on=["managerId"], right_on="id", how="inner")
    return df_final[["name"]]
```
