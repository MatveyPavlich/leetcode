# SQL
```
WITH final AS (
    SELECT
        DATE(request_at) AS 'Day',
        ROUND(
            COUNT(IF(tr.status IN('cancelled_by_client', 'cancelled_by_driver'), tr.id, NULL))
                / COUNT(tr.id),
        2)  AS 'Cancellation Rate'

    FROM trips            AS tr
        INNER JOIN users AS cl ON tr.client_id = cl.users_id
                                    AND cl.role = 'client'
                                    AND cl.banned = 'No'
        INNER JOIN users AS dr ON tr.driver_id = dr.users_id
                                AND dr.role = 'driver'
                                AND dr.banned = 'No'
    WHERE DATE(request_at) BETWEEN '2013-10-01' AND '2013-10-03'
    GROUP BY DATE(request_at)
)

SELECT *
FROM final
```
