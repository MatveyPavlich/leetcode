## My solution
```
WITH raw AS (
    SELECT
        de.name                                                     AS department,
        em.name                                                     AS employee,
        em.salary                                                   AS sal,
        DENSE_RANK()
            OVER(PARTITION BY em.departmentId ORDER BY salary DESC) AS sal_rank
    FROM employee              AS em
         INNER JOIN department AS de ON em.departmentId = de.id
),

final AS (
    SELECT
        department AS "Department",
        employee   AS "Employee",
        sal        AS "Salary"

    FROM raw
    WHERE sal_rank <=3    
)

SELECT *
FROM final
```
