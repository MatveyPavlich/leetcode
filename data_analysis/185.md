## My solution
```
WITH raw AS (
    SELECT
        de.name                                                     AS department,
        em.name                                                     AS employee,
        em.salary                                                   AS sal,
        DENSE_RANK()
            OVER(PARTITION BY em.departmentId ORDER BY salary DESC) AS sal_rank
    FROM employee              AS em
         INNER JOIN department AS de ON em.departmentId = de.id
),

final AS (
    SELECT
        department AS "Department",
        employee   AS "Employee",
        sal        AS "Salary"

    FROM raw
    WHERE sal_rank <=3    
)

SELECT *
FROM final
```

import pandas as pd

def top_three_salaries(employee: pd.DataFrame, 
                       department: pd.DataFrame) -> pd.DataFrame:

    employee['rnk'] = (employee.groupby('departmentId')['salary']
                               .rank(method = 'dense', ascending = False))

    return (employee.loc[employee.rnk <= 3]
                    .merge(department, left_on='departmentId', right_on = 'id')
                    .rename(columns = {'name_y':'Department', 'name_x':'Employee'})
                    .iloc[:,[1,2,6]])
