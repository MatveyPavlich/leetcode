```
import pandas as pd

def students_and_examinations(students, subjects, examinations):
    exam_counts = (
        examinations
        .groupby(["student_id", "subject_name"])
        .size()
        .reset_index(name="attended_exams")
    )

    df = students.merge(subjects, how="cross")

    df = df.merge(
        exam_counts,
        how="left",
        on=["student_id", "subject_name"]
    )

    df["attended_exams"] = df["attended_exams"].fillna(0).astype(int)

    return df.sort_values(by=["student_id", "subject_name"])
```
