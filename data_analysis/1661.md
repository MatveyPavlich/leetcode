```
import pandas as pd

def get_average_time(activity: pd.DataFrame) -> pd.DataFrame:
    df_start = activity[activity["activity_type"] == "start"]
    df_end = activity[activity["activity_type"] == "end"]

    df = pd.merge(df_start, df_end, how="inner", on=["machine_id", "process_id"])
    df["time"] = df["timestamp_y"] - df["timestamp_x"]
    df_final = (
        df
        .groupby("machine_id")
        .agg(
            processing_time=("time", "mean")
        )
        .sort_index()
        .reset_index()
    )
    
    df_final["processing_time"] = df_final["processing_time"].round(3)

    return df_final[["machine_id", "processing_time"]]
```
