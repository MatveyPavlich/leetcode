#SQL

```
WITH experts AS (
    SELECT
        user_id

    FROM course_completions
    GROUP BY user_id
    HAVING COUNT(course_id) >= 5
           AND AVG(course_rating) >= 4
),

pairs AS (
    SELECT
        cc.user_id,
        cc.course_name                                                                  AS first_course,
        LEAD(cc.course_name) OVER(PARTITION BY cc.user_id ORDER BY completion_date ASC) AS second_course
    
    FROM experts                       AS ex
         INNER JOIN course_completions AS cc ON ex.user_id = cc.user_id
),

final AS (
    SELECT
        first_course,
        second_course,
        COUNT(user_id) AS transition_count
    FROM  pairs
    WHERE second_course IS NOT NULL
    GROUP BY 1, 2
)

SELECT * FROM final
ORDER BY transition_count DESC, first_course ASC, second_course ASC
```
