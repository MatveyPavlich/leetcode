```
CREATE FUNCTION getNthHighestSalary(N INT) RETURNS INT
BEGIN
  RETURN (
      # Write your MySQL query statement below.
        WITH rank_calc AS (
            SELECT
                id,
                salary,
                DENSE_RANK() OVER(ORDER BY salary DESC) AS sal_rank
            
            FROM employee
        )

        SELECT DISTINCT salary AS getNthHighestSalary
        FROM rank_calc
        WHERE sal_rank = N
  );
END
```
